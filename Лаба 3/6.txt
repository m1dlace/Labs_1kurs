       static void Main(string[] args)
        {
            int[] arr = { 5, 1, 5, 3, 2, 5, -5, 0, 9, 3 };
            var min = Min(arr);
            Console.WriteLine(min);
        }
 
        private static int Min(int[] arr, int i = 0)
        {
            if (i == arr.Length - 1)
                return arr[i];
 
            return Math.Min(arr[i], Min(arr, i + 1));
        }
